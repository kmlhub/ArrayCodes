class Solution {
    public void duplicateZeros(int[] arr) {
        int sh=0;int i=0;
        for( i=0;sh+i<arr.length;++i)sh+=arr[i]==0?1:0;
        for(i=i-1;sh>0;i--)
        {
           if(i+sh<arr.length) arr[i+ sh]=arr[i];
            if(arr[i]==0)
            arr[i + --sh]=arr[i];
        }
    }
};
